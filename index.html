function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); if (data.type === 'review') { var reviewSheet = ss.getSheetByName('리뷰'); reviewSheet.appendRow([ new Date().toLocaleString('ko-KR'), data.name, data.star, data.text ]); } else { var orderSheet = ss.getSheetByName('주문목록'); orderSheet.appendRow([ new Date().toLocaleString('ko-KR'), data.name, data.phone, data.address, data.zipcode, data.qty, data.total, '미확인', '발송대기', '' ]); var subject = '[아스파라거스 주문] ' + data.name + '님 주문 접수'; var body = '새 주문이 들어왔습니다!\n\n' + '성함: ' + data.name + '\n' + '연락처: ' + data.phone + '\n' + '우편번호: ' + data.zipcode + '\n' + '배송주소: ' + data.address + '\n' + '수량: ' + data.qty + 'KG\n' + '결제금액: ' + data.total + '\n' + '주문일시: ' + new Date().toLocaleString('ko-KR'); MailApp.sendEmail('nhwoo322@naver.com', subject, body); } return ContentService .createTextOutput(JSON.stringify({ result: 'success' })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ result: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { var action = e.parameter.action; // 주문 목록 조회 if (action === 'orders') { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('주문목록'); var data = sheet.getDataRange().getValues(); var orders = []; for (var i = 1; i < data.length; i++) { if (data[i][1]) { orders.push({ row: i + 1, date: data[i][0], name: data[i][1], phone: data[i][2], address: data[i][3], zipcode: data[i][4], qty: data[i][5], total: data[i][6], payment: data[i][7], delivery: data[i][8] }); } } return ContentService .createTextOutput(JSON.stringify({ orders: orders })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ orders: [] })) .setMimeType(ContentService.MimeType.JSON); } } // 상태 업데이트 if (action === 'update') { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('주문목록'); var row = parseInt(e.parameter.row); var type = e.parameter.type; var value = e.parameter.value; // payment = H열(8번째), delivery = I열(9번째) var col = type === 'payment' ? 8 : 9; sheet.getRange(row, col).setValue(value); return ContentService .createTextOutput(JSON.stringify({ result: 'success' })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ result: 'error' })) .setMimeType(ContentService.MimeType.JSON); } } // 리뷰 조회 try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('리뷰'); var data = sheet.getDataRange().getValues(); var reviews = []; for (var i = 1; i < data.length; i++) { if (data[i][1]) { reviews.push({ date: data[i][0], name: data[i][1], star: data[i][2], text: data[i][3] }); } } return ContentService .createTextOutput(JSON.stringify(reviews)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } } function setupDashboard() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var dash = ss.getSheetByName('대시보드'); dash.clearContents(); dash.getRange('A1').setValue('아스파라거스 주문 현황'); dash.getRange('A1').setFontSize(16).setFontWeight('bold'); dash.getRange('A3').setValue('총 주문 건수'); dash.getRange('B3').setFormula('=COUNTA(주문목록!B2:B)&" 건"'); dash.getRange('A4').setValue('총 매출액'); dash.getRange('B4').setFormula('=TEXT(SUMPRODUCT((주문목록!F2:F)*40000),"#,##0")&" 원"'); dash.getRange('A5').setValue('입금 미확인'); dash.getRange('B5').setFormula('=COUNTIF(주문목록!H2:H,"미확인")&" 건"'); dash.getRange('A6').setValue('발송 대기'); dash.getRange('B6').setFormula('=COUNTIF(주문목록!I2:I,"발송대기")&" 건"'); dash.getRange('A7').setValue('발송 완료'); dash.getRange('B7').setFormula('=COUNTIF(주문목록!I2:I,"발송완료")&" 건"'); dash.getRange('A8').setValue('총 리뷰 수'); dash.getRange('B8').setFormula('=COUNTA(리뷰!B2:B)&" 개"'); dash.getRange('A9').setValue('평균 별점'); dash.getRange('B9').setFormula('=IFERROR(ROUND(AVERAGE(리뷰!C2:C),1)&" / 5.0","-")'); dash.setColumnWidth(1, 200); dash.setColumnWidth(2, 250); SpreadsheetApp.getUi().alert('대시보드 설정 완료!'); }
